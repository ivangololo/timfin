Supabase — финальный чек-лист
===============================

1. Настройка окружения
----------------------
- Создайте проект в Supabase и сохраните:
  • Project URL → переменная окружения `VITE_SUPABASE_URL`
  • anon key → переменная `VITE_SUPABASE_ANON_KEY`
- В корне фронтенда (`app/.env`) пропишите:
  ```
  VITE_SUPABASE_URL=https://<project-id>.supabase.co
  VITE_SUPABASE_ANON_KEY=<public-anon-key>
  ```
- В настройках проекта включите расширение `pgcrypto`, если оно не активировано (используется для генерации UUID).

2. Структура и политики
-----------------------
Выполните SQL-файл по разделам. Каждая секция безопасна для повторного запуска.

```sql
create or replace function generate_access_code()
returns text
language plpgsql
as $$
declare
  v_code text;
  letters constant text := 'ABCDEFGHJKLMNPQRSTUVWXYZ';
  digits constant text := '0123456789';
  i integer;
begin
  loop
    v_code := 'FG-';
    -- 4 цифры
    for i in 1..4 loop
      v_code := v_code || substr(digits, (random() * length(digits) + 1)::int, 1);
    end loop;
    v_code := v_code || '-';
    -- 4 буквы
    for i in 1..4 loop
      v_code := v_code || substr(letters, (random() * length(letters) + 1)::int, 1);
    end loop;

    exit when not exists (select 1 from access_codes where code = v_code);
  end loop;
  return v_code;
end;
$$;

create or replace function submit_access_request(p_full_name text)
returns access_requests
language plpgsql
security definer
set search_path = public
as $$
declare
  v_request access_requests;
begin
  if p_full_name is null or length(trim(p_full_name)) = 0 then
    raise exception 'Укажите имя и фамилию';
  end if;
  insert into access_requests (full_name)
  values (trim(p_full_name))
  returning * into v_request;
  return v_request;
end;
$$;

create or replace function admin_list_access_requests(admin_code text)
returns setof access_requests
language plpgsql
security definer
set search_path = public
as $$
begin
  perform app.require_admin(admin_code);
  return query
    select *
    from access_requests
    order by created_at desc;
end;
$$;

create or replace function admin_approve_access_request(admin_code text, p_request_id uuid)
returns access_requests
language plpgsql
security definer
set search_path = public
as $$
declare
  v_request access_requests;
  v_admin_id uuid;
  v_profile_id uuid;
  v_code text;
begin
  v_admin_id := app.require_owner(admin_code);

  select * into v_request
  from access_requests
  where id = p_request_id
    for update;

  if not found then
    raise exception 'Заявка не найдена';
  end if;

  if v_request.status = 'approved' then
    return v_request;
  end if;

  v_code := generate_access_code();

  insert into profiles (name, code, is_admin)
  values (v_request.full_name, v_code, false)
  on conflict (code) do update set name = excluded.name
  returning id into v_profile_id;

  insert into access_codes (code, owner_name, user_id)
  values (v_code, v_request.full_name, v_profile_id)
  on conflict (code) do update set owner_name = excluded.owner_name, user_id = excluded.user_id;

  update access_requests
     set status = 'approved',
         code = v_code,
         profile_id = v_profile_id,
         approved_by = v_admin_id,
         updated_at = now()
   where id = p_request_id
   returning * into v_request;

  return v_request;
end;
$$;

create or replace function claim_access_code(p_request_id uuid)
returns jsonb
language plpgsql
security definer
set search_path = public
as $$
declare
  v_request access_requests;
begin
  select * into v_request
  from access_requests
  where id = p_request_id
  for update;

  if not found then
    return jsonb_build_object('status', 'not_found');
  end if;

  if v_request.status <> 'approved' or v_request.code is null then
    return jsonb_build_object('status', v_request.status);
  end if;

  if v_request.delivered_at is not null then
    return jsonb_build_object('status', 'delivered');
  end if;

  update access_requests
     set delivered_at = now()
   where id = p_request_id;

  return jsonb_build_object('status', 'approved', 'code', v_request.code, 'full_name', v_request.full_name);
end;
$$;
```

```sql
-- Вспомогательные функции для работы учеников (обходят RLS через security definer)
create or replace function student_upsert_homework_submission(
  p_profile_id uuid,
  p_homework_id uuid,
  p_student_note text default null
)
returns homework_submissions
language plpgsql
security definer
set search_path = public
as $$
declare
  v_submission homework_submissions;
begin
  if not exists (select 1 from profiles where id = p_profile_id) then
    raise exception 'Профиль не найден';
  end if;

  insert into homework_submissions (homework_id, user_id, student_note)
  values (p_homework_id, p_profile_id, coalesce(p_student_note, ''))
  on conflict (homework_id, user_id) do update
    set student_note = coalesce(p_student_note, homework_submissions.student_note)
  returning * into v_submission;

  return v_submission;
end;
$$;

create or replace function student_add_submission_file(
  p_submission_id uuid,
  p_profile_id uuid,
  p_file_url text,
  p_file_type text default 'file',
  p_note text default '',
  p_role text default 'student'
)
returns homework_submission_files
language plpgsql
security definer
set search_path = public
as $$
declare
  v_submission homework_submissions;
  v_file homework_submission_files;
begin
  select *
  into v_submission
  from homework_submissions
  where id = p_submission_id
    and user_id = p_profile_id;

  if not found then
    raise exception 'Submission % not found for profile %', p_submission_id, p_profile_id;
  end if;

  insert into homework_submission_files (submission_id, uploaded_by, role, file_url, file_type, note)
  values (p_submission_id, p_profile_id, coalesce(p_role, 'student'), p_file_url, coalesce(p_file_type, 'file'), coalesce(p_note, ''))
  returning * into v_file;

  return v_file;
end;
$$;
```

### Колонка is_admin и вспомогательные функции
```sql
alter table profiles add column if not exists is_admin boolean not null default false;
alter table profiles add column if not exists is_owner boolean not null default false;
create unique index if not exists profiles_single_owner on profiles (is_owner) where is_owner;

create schema if not exists app;

create or replace function app.ensure_profile(p_code text, p_name text default null)
returns uuid
language plpgsql
security definer
set search_path = public
as $$
declare
  v_profile profiles;
begin
  insert into profiles (name, code)
  values (coalesce(p_name, 'Ученик ' || upper(trim(p_code))), upper(trim(p_code)))
  on conflict (code) do update set name = excluded.name
  returning * into v_profile;
  return v_profile.id;
end;
$$;

create or replace function app.require_admin(p_code text)
returns uuid
language plpgsql
security definer
set search_path = public
as $$
declare
  v_profile profiles;
begin
  select * into v_profile from profiles where code = upper(trim(p_code));
  if v_profile.id is null or (v_profile.is_admin is not true and v_profile.is_owner is not true) then
    raise exception 'Недостаточно прав' using errcode = '42501';
  end if;
  return v_profile.id;
end;
$$;

create or replace function app.require_owner(p_code text)
returns uuid
language plpgsql
security definer
set search_path = public
as $$
declare
  v_profile profiles;
begin
  select * into v_profile from profiles where code = upper(trim(p_code));
  if v_profile.id is null or v_profile.is_owner is not true then
    raise exception 'Недостаточно прав (требуется владелец)' using errcode = '42501';
  end if;
  return v_profile.id;
end;
$$;
```

create or replace function admin_upsert_schedule(
  admin_code text,
  p_id bigint default null,
  p_title text default null,
  p_status text default 'open',
  p_date date default null,
  p_start time default null,
  p_end time default null,
  p_format text default null,
  p_direction text default null,
  p_registration text default null,
  p_logo_url text default null,
  p_level text default null,
  p_registration_url text default null,
  p_date_meta jsonb default null,
  p_registration_start date default null,
  p_registration_end date default null
)
returns schedule
language plpgsql
security definer
set search_path = public
as $$
declare
  v_result schedule;
begin
  perform app.require_admin(admin_code);
  if p_title is null or length(trim(p_title)) = 0 then
    raise exception 'Укажите название мероприятия';
  end if;
  if p_id is null then
    insert into schedule (
      title,
      status,
      date,
      start,
      "end",
      format,
      direction,
      registration,
      logo_url,
      level,
      registration_url,
      date_meta,
      registration_start,
      registration_end
    )
    values (
      trim(p_title),
      p_status,
      p_date,
      p_start,
      p_end,
      p_format,
      p_direction,
      p_registration,
      p_logo_url,
      p_level,
      p_registration_url,
      p_date_meta,
      p_registration_start,
      p_registration_end
    )
    returning * into v_result;
  else
    update schedule
       set title = trim(p_title),
           status = p_status,
